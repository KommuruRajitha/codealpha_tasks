🧾 Java Code: StockTradingPlatform.java

java
import java.io.*;
import java.util.*;

class Stock {
    String symbol;
    double price;

    public Stock(String symbol, double price) {
        this.symbol = symbol;
        this.price = price;
    }
}

class PortfolioEntry {
    Stock stock;
    int quantity;

    public PortfolioEntry(Stock stock, int quantity) {
        this.stock = stock;
        this.quantity = quantity;
    }

    @Override
    public String toString() {
        return stock.symbol + " | Qty: " + quantity + " | Current Price: $" + stock.price;
    }
}

public class StockTradingPlatform {
    static Map<String, Stock> marketStocks = new HashMap<>();
    static List<PortfolioEntry> portfolio = new ArrayList<>();
    static Scanner sc = new Scanner(System.in);

    public static void main(String[] args) {
        loadStocks();
        loadPortfolio();
        int choice;
        do {
            System.out.println("\n=== Stock Trading Platform ===");
            System.out.println("1. View Market Stocks");
            System.out.println("2. Buy Stock");
            System.out.println("3. Sell Stock");
            System.out.println("4. View Portfolio");
            System.out.println("5. Save & Exit");
            System.out.print("Enter your choice: ");
            choice = sc.nextInt();
            switch (choice) {
                case 1 -> viewMarket();
                case 2 -> buyStock();
                case 3 -> sellStock();
                case 4 -> viewPortfolio();
                case 5 -> savePortfolio();
                default -> System.out.println("Invalid choice!");
            }
        } while (choice != 5);
    }

    static void loadStocks() {
        marketStocks.put("AAPL", new Stock("AAPL", 180.5));
        marketStocks.put("GOOG", new Stock("GOOG", 2800.75));
        marketStocks.put("TSLA", new Stock("TSLA", 750.35));
        marketStocks.put("AMZN", new Stock("AMZN", 3400.10));
    }

    static void viewMarket() {
        System.out.println("\nAvailable Stocks:");
        for (Stock stock : marketStocks.values()) {
            System.out.println(stock.symbol + " - $" + stock.price);
        }
    }

    static void buyStock() {
        System.out.print("Enter stock symbol to buy: ");
        String symbol = sc.next().toUpperCase();
        if (!marketStocks.containsKey(symbol)) {
            System.out.println("Stock not found.");
            return;
        }
        System.out.print("Enter quantity: ");
        int qty = sc.nextInt();
        for (PortfolioEntry entry : portfolio) {
            if (entry.stock.symbol.equals(symbol)) {
                entry.quantity += qty;
                System.out.println("Stock added to your existing portfolio.");
                return;
            }
        }
        portfolio.add(new PortfolioEntry(marketStocks.get(symbol), qty));
        System.out.println("Stock bought successfully.");
    }

    static void sellStock() {
        System.out.print("Enter stock symbol to sell: ");
        String symbol = sc.next().toUpperCase();
        Iterator<PortfolioEntry> it = portfolio.iterator();
        while (it.hasNext()) {
            PortfolioEntry entry = it.next();
            if (entry.stock.symbol.equals(symbol)) {
                System.out.print("Enter quantity to sell: ");
                int qty = sc.nextInt();
                if (qty >= entry.quantity) {
                    it.remove();
                    System.out.println("All shares sold.");
                } else {
                    entry.quantity -= qty;
                    System.out.println("Stock sold.");
                }
                return;
            }
        }
        System.out.println("You do not own this stock.");
    }

    static void viewPortfolio() {
        if (portfolio.isEmpty()) {
            System.out.println("Your portfolio is empty.");
            return;
        }
        System.out.println("\nYour Portfolio:");
        for (PortfolioEntry entry : portfolio) {
            System.out.println(entry);
        }
    }

    static void savePortfolio() {
        try (PrintWriter writer = new PrintWriter("portfolio.txt")) {
            for (PortfolioEntry entry : portfolio) {
                writer.println(entry.stock.symbol + "," + entry.quantity);
            }
            System.out.println("Portfolio saved successfully.");
        } catch (IOException e) {
            System.out.println("Error saving portfolio.");
        }
    }

    static void loadPortfolio() {
        File file = new File("portfolio.txt");
        if (!file.exists()) return;
        try (Scanner fileScanner = new Scanner(file)) {
            while (fileScanner.hasNextLine()) {
                String[] data = fileScanner.nextLine().split(",");
                String symbol = data[0];
                int qty = Integer.parseInt(data[1]);
                if (marketStocks.containsKey(symbol)) {
                    portfolio.add(new PortfolioEntry(marketStocks.get(symbol), qty));
                }
            }
            System.out.println("Portfolio loaded.");
        } catch (Exception e) {
            System.out.println("Error loading portfolio.");
        }
    }
}

📝 How to Run:

  1. Save the file as StockTradingPlatform.java.

  2. Compile using: javac StockTradingPlatform.java

  3. Run using: java StockTradingPlatform

  4. Portfolio data will be saved to portfolio.txt.
